# 1. Base Image: Use an official Node.js image.
FROM node:18-alpine

# 2. Set Working Directory: Create a directory inside the container.
WORKDIR /app

# 3. Copy package.json and package-lock.json:
# This takes advantage of Docker layer caching. These files don't change often.
COPY package*.json ./

# 4. Install Dependencies:
RUN npm install

# 5. Copy the rest of your application code:
COPY . .

# 6. Build the Next.js application:
RUN npm run build

# 7. Expose the port the app will run on:
EXPOSE 3000

# 8. Define the command to start the app:
CMD ["npm", "start"]
